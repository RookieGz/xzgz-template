class Node {
  constructor(val, left, right) {
    this.val = val;
    this.left = left;
    this.right = right;
  }
}

const TEST = new Node(
  1,
  new Node(2, new Node(4), new Node(5)),
  new Node(3, new Node(6), new Node(7))
);

class Node {
  constructor(val, left, right) {
    this.val = val;
    this.left = left;
    this.right = right;
  }
}

const TEST = new Node(
  1,
  new Node(2, new Node(4), new Node(5)),
  new Node(3, new Node(6), new Node(7))
);

// 层序遍历
function traversal(node) {
  const queue = [node];
  const result = [];
  while (queue.length) {
    const item = queue.shift();
    result.push(item.val);
    if (item.left) queue.push(item.left);
    if (item.right) queue.push(item.right);
  }
  return result;
}

// 前序遍历：根 -> 左 -> 右
function haedTraversal(node) {
  const stack = [];
  const result = [];
  let i = node;

  while (stack.length || i) {
    while (i) {
      stack.push(i);
      result.push(i.val);
      i = i.left;
    }

    const item = stack.pop();
    i = item.right;
  }
  return result;
}

// 中序遍历 左 -> 中 -> 右
function middleTraversal(node) {
  const stack = [];
  const result = [];
  let i = node;

  while (stack.length || i) {
    while (i) {
      stack.push(i);
      i = i.left;
    }

    const item = stack.pop();
    result.push(item.val);
    i = item.right;
  }
  return result;
}

// 后序遍历 左 -> 右 -> 中
// 加个变量，判断是否遍历了右节点
function endTraversal(node) {
  const stack = [];
  const result = [];
  let i = node;
  let prev = null;

  while (stack.length || i) {
    while (i) {
      stack.push(i);
      i = i.left;
    }

    const item = stack.pop();
    if (item.right && item.right !== prev) {
      // 如果有右侧节点，把本节点重新入栈
      stack.push(item);
      i = item.right;
    } else {
      result.push(item.val);
      prev = item
      i = null
    }
  }
  return result;
}

console.log(traversal(TEST));
console.log(haedTraversal(TEST));
console.log(middleTraversal(TEST));
console.log(endTraversal(new Node(
  3,
  new Node(9),
  new Node(4, new Node(5), new Node(7))
)));
